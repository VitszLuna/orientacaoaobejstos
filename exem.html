🧪 Exemplo 1: Objeto Fruta


let fruta = {
  nome: "Maçã",
  cor: "Vermelha",
  sabor: function() {
    return "Doce";
  }
};
alert("Nome: " + fruta.nome + "\nCor: " + fruta.cor + "\nSabor: " + fruta.sabor());

🧪 Exemplo 2: Objeto Teclado


let teclado = {
  qualidade: "Novo",
  podevender: "Disponível",  // pode ser 'Disponível' ou 'Não Disponível'
  mostrarInfo: function() {
    alert("O teclado está " + this.qualidade + " e o teclado está: " + this.podevender);
  }
};
teclado.mostrarInfo();
